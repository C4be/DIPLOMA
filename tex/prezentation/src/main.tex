\begin{frame}{Актуальность темы}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Проблема:}
      \begin{itemize}
        \item Бизнес-цели требуют быстрого отклика от аналитиков
        \item Аналитики вынуждены ждать разработчиков для получения отчётов
        \item Прослойка между аналитиком и данными становится рудиментом
      \end{itemize}
      \vspace{6pt}
      \textbf{Пример:}\\
      {\small«Какие самолёты чаще всего летали из Москвы в Санкт-Петербург в прошлом месяце с загрузкой более 80\%?» — требует нескольких JOIN и знания схемы}
    \end{column}
    \begin{column}{0.42\textwidth}
      \textbf{Факторы актуальности:}
      \begin{enumerate}
        \item Экспоненциальный рост объёмов данных
        \item Высокий порог входа в аналитику (SQL)
        \item Активное развитие LLM-моделей
        \item Ограничения существующих решений
      \end{enumerate}
      \vspace{6pt}
      \textbf{Решение:} NLIDB-интерфейс на основе RAG + LLM + агентный подход + режим самокоррекции
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Цель и задачи исследования}
  \textbf{Цель:}\\[4pt]

  \vspace{8pt}
  \textbf{Задачи:}
  \begin{enumerate}
    \item Провести оценку современных методов трансляции естественного языка в SQL
    \item Спроектировать RAG-архитектуру для работы со схемой БД (метаинформация, связи таблиц)
    \item Разработать механизм самокоррекции ИИ-агента (Self-Heal)
    \item Провести оценку качества генерации до и после самокоррекции
    \item Сравнить LLM-провайдеры: \texttt{GLM-4.7}, \texttt{Qwen3-Coder-Next}, \texttt{GigaChat-2}
    \item Реализовать микросервисную архитектуру системы
  \end{enumerate}
\end{frame}

\begin{frame}{Метрики оценки качества генерации SQL}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Execution Accuracy (EA):}
      \[
        \text{EA} = \frac{|\{q : \text{exec}(q) = \text{exec}(q^*)\}|}{|Q|}
      \]
      Доля запросов, результат выполнения которых совпал с эталоном. Совпадение результирующих множеств.

      \vspace{8pt}
      \textbf{Exact Matching (EM):}
      \[
        \text{EM} = \frac{|\{q : q = q^*\}|}{|Q|}
      \]
      Точное совпадение SQL текста.
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Exact Set Matching (ESM):}
      \[
        \text{ESM} = \frac{|\{q : R(q) \equiv R(q^*)\}|}{|Q|}
      \]
      Cравнивает структурные компоненты SQL-запроса (например, списки столбцов, условия фильтрации, ...)

      \vspace{8pt}
      \textbf{Intellect (IEA):}
      \[
        \text{IEA} = \frac{|\{q : \text{exec}(q) \approx \text{exec}(q^*)\}|}{|Q|}
      \]
      Учитывает семантически верные, но синтаксически отличающиеся запросы.

      \vspace{4pt}
      {\small Тестовый набор: \textbf{31 SQL-задача} разного уровня сложности на БД «Авиаперевозки»}
    \end{column}
  \end{columns}
\end{frame}
